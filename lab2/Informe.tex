\subsection{}
ejemplo para agregar código y como referenciarlo (ver código \ref{cod:cod1}).
\begin{lstlisting}[language={C++}, caption={Ejemplo de uso instrucción map()}, label={cod:cod1}]
int ldrPin = A0; // Pin de la fotorresistencia
int ledPin = 9;  // Pin del LED (PWM)
int valorLDR = 0;
int brillo = 0;

void setup() {
  pinMode(ledPin, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  valorLDR = analogRead(ldrPin); // Lee 0-1023
  // Mapea a 0-255 para el LED
  brillo = map(valorLDR, 0, 1023, 0, 255); 
  analogWrite(ledPin, brillo);
  Serial.println(brillo);
  delay(100);
}
\end{lstlisting}
